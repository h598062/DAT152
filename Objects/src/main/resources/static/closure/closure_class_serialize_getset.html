Serialize class instance using getters and setters and private data through closure

All output is to the web console only.

Click the button above to serialize instance of the below class: 

class Student {
    constructor(givenname, familyname) {
        // Private
        // givenname, familyname, phone
        let phone = "11223344";

        // Public
        Object.defineProperties(this, {
            "givenname": {
                "set": function(fn) { givenname = fn },
                "get": function() { return givenname },
                "enumerable": true
            },
            "familyname": {
                "set": function(ln) { familyname = ln },
                "get": function() { return familyname },
                "enumerable": true
            },
            "phone": {
                "set": function(pn) { phone = pn },
                "get": function() { return phone },
                "enumerable": true
            }
        });
    }
}
